Foresight and Hindsight Predictions

An example of code is provided to illustrate how to turn ratings into predictions of the winner of soccer outcome.

Teams ratings are produced by the following methods:

  • WinLoss

  • Colley

  • Massey

  • Elo (win version)

  • Elo (points version)

  • Keener

  • OffenseDefense

  • AccuRATE

  • GeM

Two type of predictions for the final oucome are made:

  • hindsight - predicting past matches using the ratings of teams (the first 20 matches of EPL 2018-2019)

  • foresight - predicting upcoming matches (the 3rd match week of EPL 2018-2019) using the ratings of teams for previous weeks (the first 20 matches of EPL 2018-2019)

Data (hindsight)

Soccer Data: The first 20 games of EPL (2018-2019 season)

Data (foresight)

Soccer Data: The 3rd match-week of EPL (2018-2019 season)

 3import apprate.ratings as ar
 4import pandas as pd
 5from apprate.app_sports.methods import (Predictions, predict_hindsight,
 6                                        prepare_sport_dataset,
 7                                        show_list_of_accuracy_results)
 8from apprate.application import SoccerOutcome
 9from apprate.datasets.filenames import (FILENAME_EPL_2018_2019_3RD_WEEK,
10                                        FILENAME_EPL_2018_2019_20_GAMES,
11                                        datasets_paths)
12from apprate.datasets.parameters import (COLUMNS_DICT, DATE_COL, WEEK_PERIOD,
13                                         stats)
14from apprate.datasets.parse import parse_pairs_data
15from apprate.ratings.methods import rating_systems_to_dict
16from apprate.utils.enums import ratings
17from apprate.utils.methods import print_info
18
19# rating systems
20ratings_list = [
21    ar.Winloss(normalization=False),
22    ar.Colley(),
23    ar.Massey(data_limit=10),
24    ar.Elo(version=ratings.ELOWIN, K=40, ks=400, HA=0,
25           starting_point=0),
26    ar.Elo(version=ratings.ELOPOINT, K=40, ks=400, HA=0,
27           starting_point=0),
28    ar.Keener(normalization=False),
29    ar.OffenseDefense(tol=0.0001),
30    ar.Markov(b=0.85, stats_markov_dict=stats.STATS_MARKOV_DICT),
31    ar.AccuRate(),
32]
33# convert list to dictionary
34ratings_dict = rating_systems_to_dict(ratings_list)
35# get train and test filenames
36filename, filename_test = datasets_paths(
37    FILENAME_EPL_2018_2019_20_GAMES, FILENAME_EPL_2018_2019_3RD_WEEK)
38
39# set columns dictionry and outcome
40columns_dict = COLUMNS_DICT
41outcome = SoccerOutcome()
42
43# prediction methods
44pred_methods_list = ['MLE', 'RANK']
45
46# parse train and test data
47data_train, teams_df = parse_pairs_data(filename,
48                                        parse_dates=[DATE_COL],
49                                        frequency=WEEK_PERIOD,
50                                        outcome=outcome,
51                                        columns_dict=columns_dict)
52
53# ==========================================================================
54# HINDSIGHT PREDICTIONS
55# ==========================================================================
56pred_list = []
57test_Y_list = []
58print_info("HINDSIGHT RESULTS")
59for pm in pred_methods_list:
60    for rs in ratings_list:
61        pred, test_y = predict_hindsight(
62            data_train, rs.rate_from_file(filename), outcome, pm)
63        test_Y_list.append(pred)
64        pred_list.append(test_y)
65    print_info(pm)
66    show_list_of_accuracy_results(
67        list(ratings_dict.keys()), test_Y_list, pred_list, print_predictions=True)
68
69# ==========================================================================
70# FORESIGHT PREDICTIONS
71# ==========================================================================
72
73# the test file start from the 3rd week (start_period = 3)
74data_test, teams_df = parse_pairs_data(filename_test,
75                                       parse_dates=[DATE_COL],
76                                       frequency=WEEK_PERIOD,
77                                       outcome=outcome,
78                                       start_period_from=3,
79                                       columns_dict=columns_dict)
80
81
82# concat 2 files
83data_all = pd.concat([data_train, data_test]).reset_index(drop=True)
84# prepare the dataset
85data = prepare_sport_dataset(data_season=data_all,
86                             teams_df=teams_df,
87                             rating_systems=ratings_dict,
88                             start_week=2,
89                             columns_dict=columns_dict)
90
91print_info("FORESIGHT RESULTS")
92# make foresight predictions for the 3rd week (start_from_week=3)
93results = Predictions(data, outcome, start_from_week=3,
94                      print_accuracy_report=True,
95                      print_predictions=True,
96                      print_classification_report=False).rs_pred_parallel(
97    rating_systems=ratings_dict,
98    pred_methods_list=pred_methods_list)
